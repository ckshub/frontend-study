1. 随机生成长度为n的数组，里面的的数字不要重复，取值为1 - n。

```js
写了map和用includes的方法，都说太复杂了。
最后说，只需要先填充数组，最后再打乱就行了，应该下面这么实现：

function sjsz(num){
    var ary = [];                    //创建一个空数组用来保存随机数组
    for(var i=0; i<num; i++){            //按照正常排序填充数组
        ary[i] = i+1;
    }
    ary.sort(function(){
        return 0.5-Math.random();        //返回随机正负值
    });console.log(ary)
    return ary;                    //返回数组
}
```
