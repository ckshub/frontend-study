var isValid = function (s) {
    let stack = []
    let a = "(,)"
    let b = "{,}"
    let c = "[,]"
    for (let it of s) {
        stack.push(it)
        let res = stack.slice(stack.length-2,stack.length).toString()
        if(res == a ||res == b|| res ==c ){
            stack.splice(stack.length-2,2)
        }
    }
    return stack.length == 0 ? true :false 
};

var isValid = function (s) {
    const cache = []
    const map = new Map()
    map.set("(", ")")
    map.set("[", "]")
    map.set("{", "}")
    for (let i = 0; i < s.length; i++) {
        if (map.has(s[i])) cache.push(s[i])
        else {
            if (cache.length === 0) return false
            if (map.get(cache[cache.length - 1]) === s[i]) cache.pop()
            else return false
        }
    }
    if (cache.length) return false
    return true
};
//官方
var isValid = function(s) {
    const n = s.length;
    if (n % 2 === 1) {
        return false;
    }
    const pairs = new Map([
        [')', '('],
        [']', '['],
        ['}', '{']
    ]);//map
    const stk = [];
    s.split('').forEach(ch => {
        if (pairs.has(ch)) {
            if (!stk.length || stk[stk.length - 1] !== pairs.get(ch)) {
                return false;
            }
            stk.pop();
        } 
        else {
            stk.push(ch);
        }
    });
    return !stk.length;
};
