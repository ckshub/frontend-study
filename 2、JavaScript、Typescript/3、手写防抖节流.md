##  防抖

```js
function test{
    console.log(222)
}

function debounce (fn,delay) {
    let timer = null
    return function {
        if(timer) clearTimeout(timer)    
        timer =  setTimeout(fn,delay)
    }
}
debounce(test,1000)()
```



## 节流

```js
function throttle(fn,delay){
    let valid = true //1、最开始，可以营业。
    return function {
	if(!valid) return  //3、判断，如果有了，就不营业
        valid = false  //2、开始营业
        setTimeout(
            fn 
            valid = true,100) //4、营业结束
}
```
