function Merge(pHead1, pHead2)
{
    // write code here
    let res = new ListNode(0)
    let temp = res
    //1、先确保都有，赋值是赋值当前的
    while(pHead1&&pHead2){
        if(pHead1.val<pHead2.val){
            temp.next = pHead1
            pHead1 = pHead1.next
            temp = temp.next
        }else{
            temp.next = pHead2
            pHead2 = pHead2.next
            temp = temp.next
        }
    }
    
    while(pHead1 || pHead2){
        temp.next = pHead1? pHead1 : pHead2
        temp = temp.next
        if(pHead2) pHead2 = pHead2.next
        if(pHead1) pHead1 = pHead1.next
    }

    return res.next
    
}
