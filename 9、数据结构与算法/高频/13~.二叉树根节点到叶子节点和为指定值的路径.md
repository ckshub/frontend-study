/*
 * function TreeNode(x) {
 *   this.val = x;
 *   this.left = null;
 *   this.right = null;
 * }
 */

/**
  * 
  * @param root TreeNode类 
  * @param sum int整型 
  * @return int整型二维数组
  */
function pathSum( root ,  sum ) {
    // write code here
    if(!root) return []
    return dfs(root,sum,0,[],[])
}

function dfs(root,sum,num,res,path){
    path.push(root.val);
    if ((num+root.val) === sum && !root.left && !root.right){
        res.push([...path]);
    }
    if(root.left){
        dfs(root.left, sum, num + root.val, res,path);
        path.pop();
    }
    if(root.right){
        dfs(root.right, sum, num+root.val, res,path);
        path.pop();
    }
    return res;
}
module.exports = {
    pathSum : pathSum
};
