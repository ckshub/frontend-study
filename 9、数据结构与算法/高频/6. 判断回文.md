 * @param str string字符串 待判断的字符串

function judge( str ) {
    // write code here
    let len = Math.floor(str.length / 2)
    let res =true

    for(let i = 0; i < len ;i++){
        if(str[i]==str[str.length - i -1] ){
            continue
        }else{
            res = false
        }
    }
    return res
}
module.exports = {
    judge : judge
};
