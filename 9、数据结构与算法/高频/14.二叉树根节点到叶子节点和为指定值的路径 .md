let res = false



function hasPathSum( root ,  sum) {
    // write code here
    res = false
    if(!root) {
        return res
    }
    dfs(root, sum ,0)
    return res
}

function dfs(root, sum ,num) {
//     if(res) {
//         return
//     }
    num = num + root.val
    if(!root.left && !root.right && sum == num) {
        res = true
    }
    if(root.left) {
        dfs(root.left, sum, num)
    }
    if(root.right) {
        dfs(root.right, sum, num)
    }
}
module.exports = {
    hasPathSum : hasPathSum
};
