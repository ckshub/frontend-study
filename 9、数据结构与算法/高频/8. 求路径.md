/**
  * 
  * @param m int整型 
  * @param n int整型 
  * @return int整型
  */
function uniquePaths( m ,  n ) {
    // write code here
    let dp = Array.from(new Array(m), () => new Array(n).fill(1));
    for (let i = 1; i < m; i++) {
        for (let j = 1; j < n; j++) {
            dp[i][j] = dp[i][j - 1] + dp[i - 1][j]; 
        }
    }
    return dp[m - 1][n - 1]
}
module.exports = {
    uniquePaths : uniquePaths
};
