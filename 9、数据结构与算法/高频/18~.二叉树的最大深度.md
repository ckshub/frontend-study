 * function TreeNode(x) {
 *   this.val = x;
 *   this.left = null;
 *   this.right = null;
 * }
 */

/**
  * 
  * @param root TreeNode类 
  * @return int整型
  */
function maxDepth( root ) {
    // write code here
    if(root == null) return 0
    if(!root.left && !root.right) return 1
    let res = 0
    let queue = []
    let temp = [] 
    queue.push(root)
    while (queue.length){
        res ++
        let len = queue.length
        for(let i = 0;i < len ;i++){
            let node = queue.shift()
            if(node.left) {
                temp.push(node.left)
            } 
            if(node.right){
                temp.push(node.right)
            }
        }
        queue = temp
    }
    return res 
}
module.exports = {
    maxDepth : maxDepth
};
