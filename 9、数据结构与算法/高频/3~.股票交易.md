function maxProfit( prices ) {
    // write code here
    //暴力算法
    if(prices.length == 0 || prices.length == 1){
        return 0
    }
    let stack = []
    for(let i = 0;i < prices.length;i++){
        for(let j = i+1;j<prices.length;j++){
            stack.push(prices[j] - prices[i])
        }
    }
    stack.sort(function(a,b){return b - a})
    return stack[0] > 0 ? stack[0] : 0
    //贪心算法
    let minVal = prices[0]
    let profit = 0
    for(let i = 1;i < prices.length ;i++){
        if(prices[i] < minVal) minVal = prices[i]
        profit = Math.max(profit,prices[i] - minVal)
    }
     return profit
    //动态规划
    let len = prices.length
    if(len < 2) return 0
    let minVal = prices[0]
    let profit = new Array(len).fill(0)
    for(let i = 1; i < len ; i++){
        minVal = Math.min(prices[i],minVal)
        profit[i] = Math.max(profit[i - 1],  prices[i] - minVal) 
    }
    return profit[len - 1]
    
}
module.exports = {
    maxProfit : maxProfit
};
