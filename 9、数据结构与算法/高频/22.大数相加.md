/**
 * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
 * 计算两个数之和
 * @param s string字符串 表示第一个整数
 * @param t string字符串 表示第二个整数
 * @return string字符串
 */
function solve( a ,  b ) {
    // write code here
 //取两个数字的最大长度
   let maxLength = Math.max(a.length, b.length);
    a = a.split("").reverse()
    b = b.split("").reverse()
   //定义加法过程中需要用到的变量
   let curr = 0;
   let add = 0;   //"进位"
   let sum = "";
   for(let i = 0 ; i<maxLength ; i++){
       curr = +a[i] + +b[i] + add
       add = Math.floor(curr/10)
       sum = curr%10 + sum
   }
    if(add == 1){
        sum = 1 + sum
    }
   return sum;
}
module.exports = {
    solve : solve
};
