function solve( a ,  b ) {
    // write code here
    let m = a.length
    let matrix = Array.from(new Array(m), ()=> new Array(m).fill(0))
    for(let i = 0 ;i<m; i++){
        for(let j = 0;j<m; j++){
            for(let k =0;k<m; k++ ){
               matrix[i][j] +=  a[i][k]* b[k][j] 
            }
        }
    }
    return matrix
}
module.exports = {
    solve : solve
};
